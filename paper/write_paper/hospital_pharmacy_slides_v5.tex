\section{Research Design}

\begin{frame}{County-level Empirical Setting}

\small
\begin{itemize}\setlength\itemsep{0.65em}
    \item Unit of analysis: county-by-year panel, 2012--2022.
    \item Main controlled sample: 2012--2019, excluding counties treated in 2012, with population, nighttime lights, and road-density controls.
    \item Main treatment indicator:
    \[
        DID_{ct} = \mathbf{1}\{t \geq PolicyYear_c\}.
    \]
    \item Baseline TWFE specification:
    \[
        Y_{ct}=\beta DID_{ct}+X_{ct}'\gamma+\alpha_c+\lambda_t+\varepsilon_{ct}.
    \]
    \item Preferred high-dimensional FE specification replaces year FE with province-by-year FE:
    \[
        Y_{ct}=\beta DID_{ct}+X_{ct}'\gamma+\alpha_c+\delta_{p(c)t}+\varepsilon_{ct}.
    \]
    \item Standard errors are clustered at the county level.
\end{itemize}

\vspace{0.3em}
\tiny Controls include log population, log nighttime lights, and log road density where available.

\end{frame}

\section{Synthesis}

\begin{frame}{Summary of County-level Results}

\scriptsize
\begin{itemize}\setlength\itemsep{0.35em}
    \item \textbf{TWFE count regressions are positive and stable.}
    The preferred province-year specification implies roughly 5.2 additional pharmacies after reform.

    \item \textbf{Log count estimates are small, while the extensive margin increases.}
    Log-count coefficients are imprecise; the probability of having any pharmacy rises by 2.5--3.0 percentage points.

    \item \textbf{Continuous DID suggests heterogeneity by baseline public-hospital intensity.}
    Counties with more public hospitals have larger level-count increases, but lower log-count coefficients.

    \item \textbf{CSDID and PPML are more muted.}
    CSDID average ATT and PPML count regressions do not show statistically significant average effects.

    \item \textbf{Distance results do not support a broad relocation away from hospitals.}
    Average distance and within-2km share barely change; count increases are concentrated within 2km.

    \item \textbf{Composition shifts toward chains.}
    Chain pharmacy counts and chain share increase significantly; non-chain count changes are imprecise.
\end{itemize}

\end{frame}
